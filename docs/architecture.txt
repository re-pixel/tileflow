mini-compiler/
├── src/
│   ├── compiler/            # Frontend & Middle-end (Python)
│   │   ├── ir/              # Graph IR, Tensor, and Op definitions
│   │   ├── passes/          # Optimization passes (Fusion, Tiling)
│   │   └── scheduler/       # Static memory planning & execution order
│   ├── runtime/             # Backend Execution Engine (C++)
│   │   ├── kernels/         # SIMD (AVX2) optimized math kernels
│   │   ├── memory/          # Virtual SRAM/DRAM management
│   │   └── engine.cpp       # The main execution loop (uOp dispatcher)
│   └── bindings/            # Pybind11 glue code
├── include/                 # C++ Header files (.hpp)
├── tests/                   # Unit tests (Python & C++)
│   ├── test_ir.py           # Verify graph construction
│   ├── test_kernels.cpp     # Verify SIMD math correctness
│   └── test_integration.py  # End-to-end model execution
├── examples/                # High-level scripts (e.g., building an MLP)
├── benchmarks/              # Scripts to measure latency/throughput
├── docs/                    # Architecture diagrams and design specs
├── CMakeLists.txt           # Build configuration for C++
├── setup.py                 # Build configuration for Python/Pybind11
└── requirements.txt         # Python dependencies
